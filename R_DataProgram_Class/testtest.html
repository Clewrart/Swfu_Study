任务3(10分):修改不合理分数(ifelse)。

xiugai <- function(x) {
  if (x > 1000) {
    x <- x / 100
  } else if (x > 200) {
    x <- x / 10
  } else if (x > 100) {
    x <- x - 10
  } else{
    x <- x
  }
}

for (i in 2:6) {
  sc[, i] <- sapply(sc[, i], xiugai)
}
